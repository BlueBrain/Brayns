add-bounded-planes
~~~~~~~~~~~~~~~~~~

Adds a list of axis-aligned bound limited planes.

**Params**:

        "type": "array",

            "properties": {
                "color": {
                    "description": "Geometry color",
                    "type": "array",
                    "items": {
                        "type": "number"
                    },
                    "minItems": 4,
                    "maxItems": 4
                },
                "geometry": {
                    "title": "BoundedPlane",
                    "description": "Geometry data",
                    "type": "object",
                    "properties": {
                        "bounds": {
                            "title": "Box",
                            "description": "Axis-aligned bounds to limit the plane geometry",
                            "type": "object",
                            "properties": {
                                "max": {
                                    "description": "Top front right corner XYZ",
                                    "type": "array",
                                    "items": {
                                        "type": "number"
